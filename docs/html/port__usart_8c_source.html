173void port_usart_init(uint32_t usart_id){
174

176 GPIO_TypeDef *p_port_tx = usart_arr[usart_id].p_port_tx;
177 GPIO_TypeDef *p_port_rx = usart_arr[usart_id].p_port_rx;
178 uint8_t pin_tx = usart_arr[usart_id].pin_tx;
179 uint8_t pin_rx = usart_arr[usart_id].pin_rx;

182
183 /* TO-DO alumnos: */
184 port_system_gpio_config(p_port_tx, pin_tx, GPIO_MODE_ALTERNATE, GPIO_PUPDR_PUP);
185 port_system_gpio_config(p_port_rx, pin_rx, GPIO_MODE_ALTERNATE, GPIO_PUPDR_PUP);
186 port_system_gpio_config_alternate(p_port_tx, pin_tx, alt_func_tx);
187 port_system_gpio_config_alternate(p_port_rx, pin_rx, alt_func_rx);
188
189 // Enable clock for USART peripheral.
190 RCC -> APB1ENR |= RCC_APB1ENR_USART3EN;
191
192 // Disable the USART.
193 p_usart -> CR1 &= ~USART_CR1_UE;
194
195 // Configure USART: 9600 baud rate, 8 data bits, no parity, 1 stop bit, oversampling by 16.
196 p_usart -> BRR = 0b0000011010000011;
197 p_usart -> CR1 &= ~USART_CR1_M;
198 p_usart-> CR1 &= ~(USART_CR1_PCE | USART_CR1_PS);
199 p_usart -> CR2 &= ~USART_CR2_STOP;
200 p_usart-> CR1 &= ~(USART_CR1_TE | USART_CR1_RE);
201 p_usart-> CR1 |= (USART_CR1_TE | USART_CR1_RE);
202 p_usart -> SR &= ~USART_SR_RXNE;
203 port_usart_disable_rx_interrupt(usart_id);
204 port_usart_disable_tx_interrupt(usart_id);
205
206 if (p_usart == USART3)
207 {
208 NVIC_SetPriority(USART3_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 2, 0));
209 NVIC_EnableIRQ(USART3_IRQn);
210
211 };
212
213 p_usart -> CR1 |= USART_CR1_UE;
214
215 // Reset input and output buffers.
216 _reset_buffer(usart_arr[usart_id].input_buffer, USART_INPUT_BUFFER_LENGTH);
