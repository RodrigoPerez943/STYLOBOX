102void port_buzzer_set_note_frequency(uint32_t buzzer_id, double frequency_hz) {
103 if (buzzer_id == BUZZER_0_ID) {
104 if (frequency_hz == 0) {
105 TIM3->CR1 &= ~TIM_CR1_CEN;
106 return;
107 }
108
109 TIM3->CR1 &= ~TIM_CR1_CEN;
110 TIM3->CNT = 0;
111
112 double clk = (double)SystemCoreClock;
113 double ms = 1 / frequency_hz;
114
115 double psc = (round)((clk * ms) / (65536.0) - 1.0);
116 double arr = (round)((clk * ms) / (psc + 1.0) - 1.0);
117
118 if (arr > 65535.0) {
119 psc += 1.0;
120 arr = (round)((clk * ms) / (psc + 1.0) - 1.0);
121 }
122
123 TIM3->ARR = arr;
124 TIM3->PSC = psc;
125 TIM3->CCR1 = (BUZZER_PWM_DC * (arr + 1.0));
126
127 TIM3->EGR |= TIM_EGR_UG;
128 TIM3->CCER |= TIM_CCER_CC1E;
