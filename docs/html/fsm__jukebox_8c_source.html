46bool _parse_message(char *p_message, char *p_command, char *p_param)
47{
48 char *p_token = strtok(p_message, " "); // Split the message by space
49
50 // If there's a token (command), copy it to the command variable
51 if (p_token != NULL)
52 {
53 strcpy(p_command, p_token);
54 }
55 else
56 {
57 // No command found, you might return an error or handle it as needed
58 // The USART driver of the computer sends an empty at initialization, so we will ignore it
59 return false;
60 }
61
62 // Extract the parameter (if available)
63 p_token = strtok(NULL, " "); // Get the next token
64
65 if (p_token != NULL)
66 {
67 strcpy(p_param, p_token);
68 }
69 else
70 {
71 strcpy(p_param, " "); // NO param found
72 }
73 return true;
74}

101void _execute_command(fsm_jukebox_t *p_fsm_jukebox, char *p_command, char *p_param){
102
103 // Play command
104 if (strcmp(p_command, "play") == 0) {
105 fsm_buzzer_set_action(p_fsm_jukebox->p_fsm_buzzer, PLAY);
106 }
107 // Stop command
108 else if (strcmp(p_command, "stop") == 0) {
109 fsm_buzzer_set_action(p_fsm_jukebox->p_fsm_buzzer, STOP);
110 }
111 // Pause command
112 else if (strcmp(p_command, "pause") == 0) {
113 fsm_buzzer_set_action(p_fsm_jukebox->p_fsm_buzzer, PAUSE);
114 }
115 // Speed command
116 else if (strcmp(p_command, "speed") == 0) {
117 double param = atof(p_param);
118 fsm_buzzer_set_speed(p_fsm_jukebox->p_fsm_buzzer,MAX( param , 0.1));
119 }
120 // Next song command
121 else if (strcmp(p_command, "next") == 0) {
122 _set_next_song(p_fsm_jukebox);
123 }
124 // Select command
125 else if (strcmp(p_command, "select")== 0) {
126 uint32_t melody_selected = atoi(p_param);
127 if (p_fsm_jukebox->melodies[melody_selected].melody_length != 0){
128
129 fsm_buzzer_set_action(p_fsm_jukebox->p_fsm_buzzer,STOP);
130 p_fsm_jukebox->melody_idx = melody_selected;
131 fsm_buzzer_set_melody(p_fsm_jukebox->p_fsm_buzzer, &p_fsm_jukebox->melodies[melody_selected]);
132 p_fsm_jukebox->p_melody = p_fsm_jukebox->melodies[melody_selected].p_name;
133 fsm_buzzer_set_action(p_fsm_jukebox->p_fsm_buzzer, PLAY);
134 }
135 else {
136 char msg[USART_OUTPUT_BUFFER_LENGTH];
137 sprintf(msg, "Error: Melody not found\n");
138 fsm_usart_set_out_data(p_fsm_jukebox->p_fsm_usart,msg);
139 }
140 }
141 // Info command
142 else if (strcmp(p_command, "info") == 0) {
143 char msg[USART_OUTPUT_BUFFER_LENGTH];
144 sprintf(msg, "Playing: %s\n", p_fsm_jukebox->p_melody);
145 fsm_usart_set_out_data(p_fsm_jukebox->p_fsm_usart,msg);
146 }
147 // If command not recognized
148 else {
149 char msg[USART_OUTPUT_BUFFER_LENGTH];
150 sprintf(msg, "Error: Command not found\n");
151 fsm_usart_set_out_data(p_fsm_jukebox->p_fsm_usart,msg);
152 }
